# 1월 25일 자바스크립트

```javascript
<a href="http://www.w3schools.com/" onclick="test1(); return false;">인라인</a><hr>
<a id="t1" href="http://www.w3schools.com/">고전</a><hr>
<a id="t2" href="http://www.w3schools.com/">표준</a>
<script>
function test1() {
	alert("인라인이벤트모델 버튼 클릭");	
}
function test2() {
	alert("고전이벤트모델 버튼 클릭");	
	return false;
}
function test3(e) {
	e.preventDefault();
	alert("표준이벤트모델 버튼 클릭");	
}
var link1 = document.querySelector("#t1");
var link2 = document.getElementById("t2");
link1.onclick = test2;  // 고전
link2.addEventListener("click", test3);
```

- 링크를 클릭해도 페이지 이동X



```javascript
function clickHandler() {
	var h1Tag = document.getElementsByTagName("h1")[0];
	h1Tag.addEventListener("click", displayAlert);
}
function displayAlert(e) {	
	window.alert("클릭 : " + e.pageX + ", " + e.pageY);
	window.alert("클릭 : " + e.screenX + ", " + e.screenY);
	
}
window.addEventListener("load", clickHandler);
```

- `pageX, pageY` : 페이지 기준 좌표
- `screenX, screenY` : 화면 기준 좌표



```javascript
<body onload="window.alert('반갑습니다!! 오늘은 즐거운 금욜입니다.')"> //인라인
<script>	//고전
    window.onload = function() {
         window.alert('반갑습니다!! 오늘은 즐거운 금욜입니다.');
    }
</script>
<script>	//표준
    window.addEventListener("load",function() {
         window.alert('반갑습니다!! 오늘은 즐거운 금욜입니다.')
    });
</script>
```

- load 이벤트 : 페이지 로딩이 끝난 후 발생하는 이벤트
